def commit(type='svn')
  Dir.chdir('/app1/jmeter/reports')
  case type
    when 'svn'
      puts %x{ svn add --force * && svn ci -m #{ci_message} --username #{APP_CONFIG['svn_user']} --password #{APP_CONFIG['svn_passwd']} }
    when 'git'
      %x{git add . && git commit -am #{ci_message} }
  end
end
